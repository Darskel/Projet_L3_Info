    # File Utils.rb, line 86
def ajouteBouton(boite, bouton, typeCss, w, h, x, y, callback, p1, p2)
    css = ajouteCss(typeCss)
    ajoutecssProvider(bouton, css, w, h)

    if (callback != nil) then
        bouton.signal_connect("clicked"){
            if (p1 && p2)
                callback.call(p1, p2)
            elsif (p1)
                callback.call(p1)
            else
                callback.call()
            end
        }
    end
    boite.put(bouton, x, y)
    return self
end
    ajouteCss(etat) click to toggle source

    Crée et retourne le css correspondant à l'état passé

    • etat le css demandé par l'user, 1 css sans hover, 2 avec hover

    # File Utils.rb, line 21
    def ajouteCss(etat)

        css = Gtk::CssProvider.new

        if etat == 1
            css.load(data: <<-CSS)
            button {
                opacity: 0;
                border: unset;
            }
            CSS
        elsif etat == 2
            css.load(data: <<-CSS)
            button {
                opacity: 0;
                border: unset;
            }
            button:hover {
                opacity: 0.1;
                border: 1px solid black;
            }
            CSS
        elsif etat == 3
            css.load(data: <<-CSS)
            button {
                opacity: 1;
                background: rgba(117, 190, 218, 0.0); 
                border: unset;
            }
            button:hover {
                opacity: 0.8;
                border: unset;
            }
            CSS
        end
        return css
    end

    # File Utils.rb, line 110
    def ajouteTexte(etat)
        css = Gtk::CssProvider.new

        if etat == 1
            css.load(data: <<-CSS)
            label {
                font-size: 50px;
                font-family: sans-serif;
                text-shadow: 1px 0 0 black, 0 1px 0 black, -1px 0 0 black, 0 -1px 0 black;
            }
            CSS
        elsif etat == 2
            css.load(data: <<-CSS)
            label {
                font-size: 22px;
                font-family: sans-serif; 
            }
            CSS
        elsif etat == 3
            css.load(data: <<-CSS)
            label {
                font-size: 34px;
                font-family: sans-serif;
                text-shadow: 1px 0 0 black, 0 1px 0 black, -1px 0 0 black, 0 -1px 0 black;
            }
            CSS
        elsif etat == 4
            css.load(data: <<-CSS)
            label {
                font-size: 34px;
                font-family: sans-serif; 
                color: grey;
            }
            CSS
        end
        return css
    end

def ajoutecssProvider(bouton, css, width, height)
    bouton.style_context.add_provider(css, Gtk::StyleProvider::PRIORITY_USER)
    bouton.set_size_request(width, height)
    return self
end
